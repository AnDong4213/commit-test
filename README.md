```java
列出所有标签: git tag
查看标签信息：git show v1.0

# 创建一个轻量级标签
git tag v1.0

# 创建一个附注标签
git tag -a v1.0 -m "Version 1.0"

# 创建一个签名标签（需要GPG密钥）
git tag -s v1.0 -m "Signed version 1.0"

# 推送所有标签到远程仓库
git push origin --tags

# 只推送特定的标签
git push origin v1.0

git tag -d v1.0

git push origin :refs/tags/v1.0
```
